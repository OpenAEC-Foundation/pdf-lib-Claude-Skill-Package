# Error Reproduction and Fix Examples

## Example 1: Loading an Encrypted PDF (Owner Password)

### Error Reproduction

```typescript
import { PDFDocument } from 'pdf-lib';
import fs from 'fs';

// PDF with owner-password restrictions (no-print, no-copy)
const pdfBytes = fs.readFileSync('restricted-document.pdf');

// This THROWS: encryption error
const pdfDoc = await PDFDocument.load(pdfBytes);
// Error: "Failed to parse PDF document (Input document to PDFParser is encrypted)"
```

### Fix

```typescript
import { PDFDocument } from 'pdf-lib';
import fs from 'fs';

const pdfBytes = fs.readFileSync('restricted-document.pdf');

// Bypass the encryption check
const pdfDoc = await PDFDocument.load(pdfBytes, {
  ignoreEncryption: true,
});

// Verify content is accessible
console.log(`Pages: ${pdfDoc.getPageCount()}`);
console.log(`Encrypted flag: ${pdfDoc.isEncrypted}`);

// Proceed with modifications
const page = pdfDoc.getPages()[0];
page.drawText('Added text', { x: 50, y: 50, size: 12 });

const outputBytes = await pdfDoc.save();
fs.writeFileSync('modified-output.pdf', outputBytes);
```

## Example 2: Loading a User-Password Encrypted PDF

### Error Reproduction

```typescript
import { PDFDocument } from 'pdf-lib';

// PDF with user-password encryption (content is encrypted)
const pdfBytes = fs.readFileSync('password-protected.pdf');

// Option A: Without ignoreEncryption — throws immediately
const pdfDoc = await PDFDocument.load(pdfBytes);
// Error: encryption error

// Option B: With ignoreEncryption — loads but content is garbage
const pdfDoc2 = await PDFDocument.load(pdfBytes, {
  ignoreEncryption: true,
});
// pdfDoc2.isEncrypted === true
// Content operations produce corrupted/empty results
```

### Fix: Pre-process with External Tool

```typescript
import { PDFDocument } from 'pdf-lib';
import { execSync } from 'child_process';
import fs from 'fs';

// Step 1: Decrypt using qpdf (must be installed on system)
execSync('qpdf --password=mypassword --decrypt input.pdf decrypted.pdf');

// Step 2: Load the decrypted PDF normally
const decryptedBytes = fs.readFileSync('decrypted.pdf');
const pdfDoc = await PDFDocument.load(decryptedBytes);

// Content is now fully accessible
const pages = pdfDoc.getPages();
console.log(`Loaded ${pages.length} pages from decrypted PDF`);
```

## Example 3: Loading a Malformed PDF (Lenient Mode)

### Scenario

```typescript
import { PDFDocument } from 'pdf-lib';

// PDF generated by buggy software with invalid object streams
const corruptedBytes = fs.readFileSync('malformed.pdf');

// Default: silently skips invalid objects
const pdfDoc = await PDFDocument.load(corruptedBytes);
// Loads successfully but some content may be missing
// where invalid objects were skipped
```

## Example 4: Loading a Malformed PDF (Strict Mode for Debugging)

### Error Reproduction

```typescript
import { PDFDocument } from 'pdf-lib';

const corruptedBytes = fs.readFileSync('malformed.pdf');

// Strict mode: throws on first invalid object
const pdfDoc = await PDFDocument.load(corruptedBytes, {
  throwOnInvalidObject: true,
});
// Error: details about which object is invalid
```

### Fix: Diagnose Then Load Leniently

```typescript
import { PDFDocument } from 'pdf-lib';

const corruptedBytes = fs.readFileSync('malformed.pdf');

// Step 1: Identify the problem
try {
  await PDFDocument.load(corruptedBytes, {
    throwOnInvalidObject: true,
  });
} catch (error) {
  console.error('Found invalid object:', error.message);
  // Log the error for analysis
}

// Step 2: Load leniently for production use
const pdfDoc = await PDFDocument.load(corruptedBytes, {
  throwOnInvalidObject: false, // default
});

// Step 3: Verify content integrity
const pageCount = pdfDoc.getPageCount();
console.log(`Loaded ${pageCount} pages (some content may be missing)`);
```

## Example 5: Metadata Preservation

### Error Reproduction

```typescript
import { PDFDocument } from 'pdf-lib';

const pdfBytes = fs.readFileSync('legal-document.pdf');

// Default load: metadata is overwritten
const pdfDoc = await PDFDocument.load(pdfBytes);

// These are now OVERWRITTEN:
console.log(pdfDoc.getProducer());
// "pdf-lib (https://github.com/Hopding/pdf-lib)" — NOT the original producer
console.log(pdfDoc.getCreationDate());
// Current timestamp — NOT the original creation date
```

### Fix

```typescript
import { PDFDocument } from 'pdf-lib';

const pdfBytes = fs.readFileSync('legal-document.pdf');

// Preserve all original metadata
const pdfDoc = await PDFDocument.load(pdfBytes, {
  updateMetadata: false,
});

// Original metadata is intact
console.log(pdfDoc.getProducer());   // Original producer software
console.log(pdfDoc.getCreationDate()); // Original creation date
console.log(pdfDoc.getAuthor());       // Original author
```

## Example 6: Restricted Form Filling

### Error Reproduction

```typescript
import { PDFDocument } from 'pdf-lib';

// PDF form with owner-password restriction on form filling
const formBytes = fs.readFileSync('restricted-form.pdf');

// Throws encryption error
const pdfDoc = await PDFDocument.load(formBytes);
```

### Fix

```typescript
import { PDFDocument } from 'pdf-lib';

const formBytes = fs.readFileSync('restricted-form.pdf');

// Bypass restriction check
const pdfDoc = await PDFDocument.load(formBytes, {
  ignoreEncryption: true,
});

// Fill the form normally
const form = pdfDoc.getForm();
const fields = form.getFields();

// List all field names for reference
fields.forEach(field => {
  console.log(`${field.constructor.name}: "${field.getName()}"`);
});

// Fill specific fields
form.getTextField('applicant_name').setText('Jane Smith');
form.getTextField('date').setText('2024-01-15');

const outputBytes = await pdfDoc.save();
fs.writeFileSync('filled-form.pdf', outputBytes);
```

## Example 7: Invalid Input Type

### Error Reproduction

```typescript
import { PDFDocument } from 'pdf-lib';

// WRONG: passing a file path
const pdfDoc = await PDFDocument.load('/path/to/document.pdf');
// Error: not valid PDF bytes

// WRONG: passing a URL
const pdfDoc2 = await PDFDocument.load('https://example.com/doc.pdf');
// Error: not valid PDF bytes
```

### Fix: Node.js

```typescript
import { PDFDocument } from 'pdf-lib';
import fs from 'fs';

// Read file into bytes first
const pdfBytes = fs.readFileSync('/path/to/document.pdf');
const pdfDoc = await PDFDocument.load(pdfBytes);
```

### Fix: Browser

```typescript
import { PDFDocument } from 'pdf-lib';

// Fetch and convert to ArrayBuffer
const response = await fetch('https://example.com/doc.pdf');
const pdfBytes = await response.arrayBuffer();
const pdfDoc = await PDFDocument.load(pdfBytes);
```

### Fix: File Input (Browser)

```typescript
import { PDFDocument } from 'pdf-lib';

// From <input type="file">
const fileInput = document.getElementById('pdfInput') as HTMLInputElement;
const file = fileInput.files![0];
const pdfBytes = await file.arrayBuffer();
const pdfDoc = await PDFDocument.load(pdfBytes);
```

## Example 8: Maximum Compatibility Loading

### Pattern for Unknown PDFs

```typescript
import { PDFDocument } from 'pdf-lib';

async function loadPdfSafely(
  pdfBytes: Uint8Array | ArrayBuffer
): Promise<PDFDocument> {
  try {
    // First attempt: standard load
    return await PDFDocument.load(pdfBytes);
  } catch (error) {
    const message = (error as Error).message;

    if (message.includes('encrypt')) {
      // Retry with encryption bypass
      console.warn('PDF is encrypted, bypassing encryption check');
      return await PDFDocument.load(pdfBytes, {
        ignoreEncryption: true,
        updateMetadata: false,
      });
    }

    // For other errors, try maximum leniency
    console.warn('PDF loading failed, trying lenient mode:', message);
    return await PDFDocument.load(pdfBytes, {
      ignoreEncryption: true,
      throwOnInvalidObject: false,
      updateMetadata: false,
      capNumbers: true,
    });
  }
}
```

## Example 9: capNumbers for Extreme Coordinates

### Error Reproduction

```typescript
import { PDFDocument } from 'pdf-lib';

// PDF with extreme floating-point values in content streams
const pdfBytes = fs.readFileSync('cad-drawing.pdf');
const pdfDoc = await PDFDocument.load(pdfBytes);

const page = pdfDoc.getPages()[0];
const { width, height } = page.getSize();
console.log(width, height); // NaN or Infinity — broken dimensions
```

### Fix

```typescript
import { PDFDocument } from 'pdf-lib';

const pdfBytes = fs.readFileSync('cad-drawing.pdf');
const pdfDoc = await PDFDocument.load(pdfBytes, {
  capNumbers: true,
});

const page = pdfDoc.getPages()[0];
const { width, height } = page.getSize();
console.log(width, height); // Finite, usable values
```
